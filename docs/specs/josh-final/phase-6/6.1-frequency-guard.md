// Invitation cadence constants.

export const INVITATION_EXPIRY_HOURS = 48;

export const INVITATION_WEEKLY_CAP = 3;
export const INVITATION_BACKOFF_THRESHOLD = 5; // backoff_count >= this → suppressed
export const INVITATION_WEEK_ROLLING_DAYS = 7;
export const MAX_CLUSTER_SIZE = 10;
export const CLUSTER_MIN_PAIRWISE_SCORE = 0.65;

export const INVITATION_IDEMPOTENCY_KEY = (
  userId: string,
  activityKey: string,
  isoWeek: string,
): string => `invitation:${userId}:${activityKey}:${isoWeek}`;

### File: `packages/core/src/invitation/frequency-guard.ts`

```ts
export type FrequencyGuardResult =
  | { eligible: true }
  | {
      eligible: false;
      reason:
        | "weekly_cap"
        | "cooldown"
        | "backoff_suppressed"
        | "eligibility_gate";
    };

export async function checkInvitationEligibility(
  userId: string,
): Promise<FrequencyGuardResult>;
```

Checks in order — fail fast on first ineligible condition:

**1. Eligibility gate**

Call `evaluateEligibility({ userId, action_type: 'can_receive_invitation' })`.
If ineligible: return `{ eligible: false, reason: 'eligibility_gate' }`.

**2. Cooldown**

Load `users.last_invited_at` for the user.
If `last_invited_at` is not null and `now() - last_invited_at < INVITATION_COOLDOWN_DAYS days`:
return `{ eligible: false, reason: 'cooldown' }`.

**3. Weekly cap**

Load `users.invitation_week_start` and `users.invitation_count_this_week`.

- If `invitation_week_start` is null, or `now() - invitation_week_start > INVITATION_WEEK_ROLLING_DAYS days`: the rolling window has expired — user has zero invitations this week. Pass this check.
- Otherwise: if `invitation_count_this_week >= INVITATION_WEEKLY_CAP`: return `{ eligible: false, reason: 'weekly_cap' }`.

**4. Backoff suppression**

Load `users.invitation_backoff_count`.
If `invitation_backoff_count >= INVITATION_BACKOFF_THRESHOLD`:
return `{ eligible: false, reason: 'backoff_suppressed' }`.

**5. Eligible**

return `{ eligible: true }`.

All checks are pure reads. This function does not update any state.

## Hard Rules for This Ticket

- All numeric thresholds live in `constants.ts` only — never hardcoded inline
- This function never mutates state — no DB writes, no session updates
- The weekly cap rolling window uses `invitation_week_start`, not a fixed calendar week. Rolling window resets when `now() - invitation_week_start > 7 days`.
- Check order matters: eligibility gate first, then cooldown, then cap, then backoff. A user on safety hold is caught at step 1 and never reaches cooldown.
